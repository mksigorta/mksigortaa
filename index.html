☁️ Supabase Bulut Veritabanı
Bağlı Değil

Supabase bağlantısı kurulduğunda tüm veriler gerçek zamanlı olarak buluta kaydedilir. Birden fazla cihazdan erişim sağlanır. Önce Supabase projenizde mk_database tablosunu oluşturun.

📋 Supabase SQL Editor'de Çalıştırın:
-- Ana veri tablosu
create table if not exists mk_database (
  id integer primary key default 1,
  data jsonb not null default '{}',
  updated_at timestamptz default now()
);
create unique index if not exists mk_database_single on mk_database (id);
alter table mk_database disable row level security;

-- Yedek tablosu
create table if not exists mk_backups (
  id text primary key,
  tarih text,
  saat text,
  data jsonb not null default '{}',
  created_at timestamptz default now()
);
alter table mk_backups disable row level security;

-- Ayarlar tablosu (şifre, kullanıcılar vb. cihazlar arası senkronizasyon)
create table if not exists mk_settings (
  key text primary key,
  value text not null default '',
  updated_at timestamptz default now()
);
alter table mk_settings disable row level security;
